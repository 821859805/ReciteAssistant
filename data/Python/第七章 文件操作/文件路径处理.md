**os.path（传统方式）：**
```python
import os

os.path.join("dir", "file.txt")     # 'dir/file.txt'
os.path.abspath("file.txt")         # 绝对路径
os.path.dirname("/a/b/c.txt")       # '/a/b'
os.path.basename("/a/b/c.txt")      # 'c.txt'
os.path.splitext("file.tar.gz")     # ('file.tar', '.gz')
os.path.exists("file.txt")          # 是否存在
os.path.isfile("file.txt")          # 是否是文件
os.path.isdir("directory")          # 是否是目录
os.path.getsize("file.txt")         # 文件大小（字节）
```

**pathlib.Path（推荐，Python 3.4+）：**
```python
from pathlib import Path

# 创建路径
p = Path("data") / "subdir" / "file.txt"

# 路径属性
p.name          # 'file.txt'
p.stem          # 'file'
p.suffix        # '.txt'
p.suffixes      # ['.txt']
p.parent        # Path('data/subdir')
p.parents       # 所有父目录
p.absolute()    # 绝对路径
p.resolve()     # 解析符号链接后的绝对路径

# 路径判断
p.exists()      # 是否存在
p.is_file()     # 是否是文件
p.is_dir()      # 是否是目录

# 文件操作
p.read_text(encoding="utf-8")       # 读取文本
p.write_text("hello", encoding="utf-8")  # 写入文本
p.read_bytes()                       # 读取字节
p.write_bytes(b"hello")              # 写入字节

# 目录操作
Path("new_dir").mkdir(parents=True, exist_ok=True)
p.unlink()                           # 删除文件
p.rename("new_name.txt")             # 重命名

# 遍历与搜索
list(Path(".").iterdir())            # 列出目录内容
list(Path(".").glob("*.py"))         # 匹配当前目录
list(Path(".").rglob("*.py"))        # 递归匹配
```

**临时文件：**
```python
import tempfile

# 创建临时文件
with tempfile.NamedTemporaryFile(mode='w', suffix='.txt', delete=False) as f:
    f.write("临时数据")
    print(f.name)  # 临时文件路径

# 创建临时目录
with tempfile.TemporaryDirectory() as tmpdir:
    print(tmpdir)  # 退出时自动删除
```
